// ── DIODE (Nâng cấp độ chi tiết cực cao) ─────────────────────────── function buildDiode(cx, cz, angle) { const g = new THREE.Group(); g.position.set(cx, 0, cz); g.rotation.y = angle || 0; const len = 2.8, rad = 0.45, S = 24, h = 0.7; // Độ cao so với board // 1. Thân Diode (Chất liệu thủy tinh bóng mờ 1N4148) const bodyMat = new THREE.MeshStandardMaterial({ color: 0x771a0a, roughness: 0.1, metalness: 0.2, transparent: true, opacity: 0.9, emissive: 0x330600, emissiveIntensity: 0.4 }); const body = cyl(rad, rad, len, S, bodyMat); body.rotation.z = Math.PI / 2; body.position.set(0, h, 0); g.add(body); // 2. Vạch đánh dấu cực âm (Cathode Band) - Sát mép thực tế const bandMat = mat(0xeeeeee, 0.4, 0.1); const band = cyl(rad + 0.02, rad + 0.02, 0.35, S, bandMat); band.rotation.z = Math.PI / 2; band.position.set(len * 0.3, h, 0); // Vị trí vạch bạc g.add(band); // 3. Hai đầu bọc kim loại (Internal junction) [-len/2, len/2].forEach(x => { const cap = cyl(rad - 0.05, rad - 0.05, 0.15, S, M.leg); cap.rotation.z = Math.PI / 2; cap.position.set(x, h, 0); g.add(cap); }); // 4. Chân linh kiện (Leads) - Uốn cong cơ khí const leadRadius = 0.08; const leadExt = 1.8; // Độ dài chân chạy ngang trước khi cắm xuống [-1, 1].forEach(side => { // Đoạn nằm ngang const hLead = box(leadExt, leadRadius, leadRadius, M.leg); hLead.position.set(side * (len/2 + leadExt/2), h, 0); g.add(hLead); // Khớp nối bo tròn tại góc uốn const knee = new THREE.Mesh(new THREE.SphereGeometry(leadRadius, 8, 8), M.leg); knee.position.set(side * (len/2 + leadExt), h, 0); g.add(knee); // Đoạn cắm xuống board const vLead = box(leadRadius, h, leadRadius, M.leg); vLead.position.set(side * (len/2 + leadExt), h/2, 0); g.add(vLead); // Mối hàn (Solder) tại điểm tiếp xúc PCB solder(g, side * (len/2 + leadExt), 0, 1.2); }); // 5. Hiệu ứng Forward-Bias (Ánh sáng khi có dòng chạy qua) const biasLight = new THREE.PointLight(0xff3300, 0.8, 5); biasLight.position.set(0, h + 0.5, 0); g.add(biasLight); scene.add(g); return g; } // ── TRIỂN KHAI DIODE VÀO CÁC VỊ TRÍ CHIẾN THUẬT ─────────────── // (Thêm đoạn này vào phần BUILD CITY của bạn) [ [-19, 6, 0], [-19, -6, Math.PI], [-19, 12, 0], [-19, -12, Math.PI], [17, 6, 0], [17, -6, Math.PI], [17, 12, 0], [17, -12, Math.PI], [-6, -19, Math.PI/2], [6, -19, Math.PI/2], [-12, -19, Math.PI/2], [-6, 17, -Math.PI/2], [6, 17, -Math.PI/2], [12, 17, -Math.PI/2] ].forEach(([x, z, a]) => buildDiode(x, z, a));
